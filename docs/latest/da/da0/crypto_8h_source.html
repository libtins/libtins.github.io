388  template<typename ForwardIterator>
389  RC4Key::RC4Key(ForwardIterator start, ForwardIterator end) {
390  for(size_t i = 0; i < data_size; ++i)
391  data[i] = i;
392  size_t j = 0;
393  ForwardIterator iter = start;
394  for(size_t i = 0; i < data_size; ++i) {
395  j = (j + data[i] + *iter++) % 256;
396  if(iter == end)
397  iter = start;
398  std::swap(data[i], data[j]);
399  }
400  }

402  template<typename ForwardIterator, typename OutputIterator>
403  void rc4(ForwardIterator start, ForwardIterator end, RC4Key &key, OutputIterator output) {
404  size_t i = 0, j = 0;
405  while(start != end) {
406  i = (i + 1) % RC4Key::data_size;
407  j = (j + key.data[i]) % RC4Key::data_size;
408  std::swap(key.data[i], key.data[j]);
409  *output++ = *start++ ^ key.data[(key.data[i] + key.data[j]) % RC4Key::data_size];
410  }
411  }
412 }
