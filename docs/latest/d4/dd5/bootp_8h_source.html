269  void chaddr(const HWAddress<n>& new_chaddr) {
270  size_t copy_threshold = std::min(n, sizeof(bootp_.chaddr));
271  for (size_t i = 0; i < copy_threshold; ++i) {
272  if (i < copy_threshold) {
273  bootp_.chaddr[i] = new_chaddr[i];
