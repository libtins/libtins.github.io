72 inline uint16_t do_change_endian(uint16_t data) {
73  return ((data & 0xff00) >> 8) | ((data & 0x00ff) << 8);

81 inline uint32_t do_change_endian(uint32_t data) {
82  return (((data & 0xff000000) >> 24) | ((data & 0x00ff0000) >> 8) |
83  ((data & 0x0000ff00) << 8) | ((data & 0x000000ff) << 24));
84 }
